   120Ôćĺ// ---- Wilayah extraction from text ----
   121Ôćĺ// Handles both "Provinsi : Jawa Barat" (label: value) and
   122Ôćĺ// "Kec. Kraton, Kab. Pasuruan, Prov. Jawa Timur" (comma-separated single line)
   123Ôćĺexport function extractWilayahFromText(text: string): Partial<PKHFormData> {
   124Ôćĺ  const result: Partial<PKHFormData> = {}
   125Ôćĺ
   126Ôćĺ  // Strategy: find each field label and capture up to the next comma+label or end of line
   127Ôćĺ  // This handles "Kec. Kraton, Kab. Pasuruan, Prov. Jawa Timur" Ôćĺ Kecamatan=Kraton, Kab=Pasuruan, Prov=Jawa Timur
   128Ôćĺ  // Labels: "provinsi" or "prov." (with period). Use lookahead for separator to handle

   133Ôćĺ    kecamatan: /\b(?:kecamatan|kec\.)(?=\s|:|-|\n|,\s)/i,

   135Ôćĺ  }
   136Ôćĺ
   137Ôćĺ  // For each field, find the label, then capture the value up to the next comma+label or newline
   138Ôćĺ  for (const [field, labelRegex] of Object.entries(fieldPatterns)) {
   139Ôćĺ    const labelMatch = text.match(labelRegex)
   140Ôćĺ    if (!labelMatch) continue

   146Ôćĺ      value = value

   154Ôćĺ  }

   162Ôćĺ        result.kelurahan = val

   164Ôćĺ    }
   165Ôćĺ  }
   166Ôćĺ

   169Ôćĺ  if (triwulanMatch) {
   170Ôćĺ    const triwulan = parseInt(triwulanMatch[1], 10)
   171Ôćĺ    const tahun = parseInt(triwulanMatch[2], 10)
   172Ôćĺ    result.triwulan = triwulan
   173Ôćĺ    result.tahun = tahun
   174Ôćĺ    result.periode = `TRIWULAN ${triwulan} TAHUN ${tahun}`
   175Ôćĺ  } else {
   176Ôćĺ    const periodeMatch = text.match(/\bperiode\s*[:\-]?\s*([^\n\r]+)/i)
   177Ôćĺ    if (periodeMatch && periodeMatch[1]) {
   178Ôćĺ      const val = periodeMatch[1].trim().replace(/\s+/g, ' ').substring(0, 60)
   179Ôćĺ      result.periode = val
   180Ôćĺ    }
   181Ôćĺ  }
   182Ôćĺ
   183Ôćĺ  // NPSN (8-digit school code)

   185Ôćĺ  if (npsnMatch) result.npsn = npsnMatch[1]
   186Ôćĺ
   187Ôćĺ  // Nama Sekolah / Posyandu / Wilayah
   188Ôćĺ  const sekolahMatch = text.match(/\bnama\s+sekolah\s*[:\-]?\s*([^\n\r|]+?)(?=\s+(?:alamat|kec|kab|prov|npsn|nik)\b|\n|\r|$)/i)

   190Ôćĺ    result.namaSekolah = sekolahMatch[1].trim().replace(/\s+/g, ' ').substring(0, 100)

   195Ôćĺ  if (alamatMatch && alamatMatch[1]) {

   207Ôćĺ  const nipLineMatch = text.match(/\bNIP\.?\s*[:\-]?\s*([\d\sA-Za-z]{16,24})/i)
   208Ôćĺ  if (nipLineMatch && nipLineMatch[1]) {

   212Ôćĺ      result.signerNIP = cleaned
   213Ôćĺ    }
   214Ôćĺ  }
   215Ôćĺ
   216Ôćĺ  // Signer name ÔÇö look for a capitalized name between role label and NIP
   217Ôćĺ  // The name may have stamp text overlapping (garbled), so find the best name-like pattern
   218Ôćĺ  if (signerRoleMatch) {

   220Ôćĺ    const nipIdx = text.indexOf('NIP', roleEnd)
   221Ôćĺ    if (nipIdx > roleEnd) {
   222Ôćĺ      const between = text.substring(roleEnd, nipIdx)
   223Ôćĺ      // Match an Indonesian name: optional honorific + 2+ capitalized words + optional academic titles
   224Ôćĺ      // e.g. "H. MOH. HANSAN, S.Pd.I, M.Pd"
   225Ôćĺ      // Find ALL matches and pick the best name candidate
   226Ôćĺ      const namePattern = /(?:(?:H\.|Hj\.|Drs\.|Dr\.|KH\.)\s+)?[A-Z][A-Za-z.']+(?:\s+[A-Z][A-Za-z.']*){1,6}(?:,?\s*(?:S\.Pd\.I|S\.Pd|S\.Ag|S\.Si|S\.Sos|S\.Kom|M\.Pd|M\.Si|M\.Sc|M\.A|M\.Pd\.I|M\.Ag|B\.A|Ph\.D))?/g

   229Ôćĺ      // (too many single-letter words = stamp overlap noise)
   230Ôćĺ      const headerWords = /^(Form|Dokumen|Status|Periode|Alamat|NPSN|Nama|Sekolah|Kementerian|Program|Direktorat|Catatan|Mengetahui|Triwulan|Tahun|Verifikasi|Komitmen|Pendidikan|Kesehatan|Kesejahteraan|Sosial|Pendamping|Koordinator|Kepala|Desa|Lurah|Provinsi|Kabupaten|Kecamatan|Kelurahan|Bentuk|Tingkat|Hari|Efektif|Alpa|Izin|Sakit|Jml|Persentase|Keterangan|Januari|Februari|Maret|April|Mei|Juni|Juli|Agustus|September|Oktober|November|Desember|Tangan|Tanda|Diverifikasi|Ditandatangani|Secara|Elektronik|Menggunakan|Sertifikat|Badan|Siber|Sandi|Negara|Keaslian|Dapat|Melalui|Validasi|Disahkan|BSrE|PKH|RI)$/i

   232Ôćĺ        .map((m) => {

   244Ôćĺ          return { name: trimmed, score }
   245Ôćĺ        })
   246Ôćĺ        .filter((s) => s.score > 0 && s.name.length > 5)
   247Ôćĺ        .sort((a, b) => b.score - a.score)
   248Ôćĺ      if (scored.length > 0) {

   258Ôćĺ    if (val.length > 3 && val.length < 50) result.facilitator = val
   259Ôćĺ  }
   260Ôćĺ
   261Ôćĺ  return result
   262Ôćĺ}

   264Ôćĺ// ---- Form type detection from free text ----
   265Ôćĺexport function detectFormTypeFromText(text: string): FormType {
   266Ôćĺ  const lower = text.toLowerCase()
   267Ôćĺ
   268Ôćĺ  const eduScore =
   269Ôćĺ    (lower.match(/pendidikan/g)?.length || 0) * 3 +
   270Ôćĺ    (lower.match(/sekolah/g)?.length || 0) * 2 +
   271Ôćĺ    (lower.match(/kelas\b/g)?.length || 0) +

   274Ôćĺ    (lower.match(/nisn|npsn/g)?.length || 0) * 2
   275Ôćĺ
   276Ôćĺ  const healthScore =
   277Ôćĺ    (lower.match(/kesehatan/g)?.length || 0) * 2 +
   278Ôćĺ    (lower.match(/posyandu/g)?.length || 0) * 3 +
   279Ôćĺ    (lower.match(/pemeriksaan/g)?.length || 0) * 2 +
   280Ôćĺ    (lower.match(/berat\s*badan|\bbb\b/g)?.length || 0) +
   281Ôćĺ    (lower.match(/tinggi\s*badan|\btb\b/g)?.length || 0) +
   282Ôćĺ    (lower.match(/keluarga\s+sehat/g)?.length || 0) * 2
   283Ôćĺ
   284Ôćĺ  const socialScore =
   285Ôćĺ    (lower.match(/kesejahteraan\s+sosial/g)?.length || 0) * 3 +
   286Ôćĺ    (lower.match(/\bbantuan\b/g)?.length || 0) * 2 +
   287Ôćĺ    (lower.match(/penerima\s+bantuan/g)?.length || 0) * 2 +
   288Ôćĺ    (lower.match(/jumlah\s*bantuan/g)?.length || 0) +
   289Ôćĺ    (lower.match(/\baktif\b|\btidak\s+aktif\b/g)?.length || 0)
   290Ôćĺ
   291Ôćĺ  if (eduScore >= healthScore && eduScore >= socialScore && eduScore > 0) return 'education'
   292Ôćĺ  if (healthScore >= socialScore && healthScore > 0) return 'health'
   293Ôćĺ  if (socialScore > 0) return 'social'
   294Ôćĺ  return 'social'
   295Ôćĺ}

   309Ôćĺexport function parseAttendanceFromText(
   310Ôćĺ  text: string,
   311Ôćĺ  months: string[]
   312Ôćĺ): { hariEfektif: number[]; rows: ParsedAttendanceRow[] } {

   314Ôćĺ

   323Ôćĺ  }
   324Ôćĺ  // Pad to 3 if fewer found
   325Ôćĺ  while (hariEfektifVals.length < 3) {
   326Ôćĺ    hariEfektifVals.push(hariEfektifVals[hariEfektifVals.length - 1] || 22)
   327Ôćĺ  }
   328Ôćĺ
   329Ôćĺ  // 2) Extract attendance data rows
   330Ôćĺ  // A valid attendance line has 15 numbers: [A I S JML %] ├Ś 3 months
   331Ôćĺ  // Percent values have "%" suffix. Line may also contain "Hadir"/"Tidak Hadir" + name.
   332Ôćĺ  // Pattern: "0 0 1 21 95% 0 0 0 20 100% 0 1 0 21 95% Hadir ABDUL BASRI"
   333Ôćĺ  const rows: ParsedAttendanceRow[] = []
   334Ôćĺ  // Match 3 groups of (num num num num num%)
   335Ôćĺ  const attRegex = /(\d+)\s+(\d+)\s+(\d+)\s+(\d+)\s+(\d+)\s*%\s+(\d+)\s+(\d+)\s+(\d+)\s+(\d+)\s+(\d+)\s*%\s+(\d+)\s+(\d+)\s+(\d+)\s+(\d+)\s+(\d+)\s*%/
   336Ôćĺ
   337Ôćĺ  for (const line of lines) {
   338Ôćĺ    // Skip header lines
   339Ôćĺ    if (/^(hari\s*efektif|alpa|izin|sakit|jml|no|nik|nama|form|program|periode|catatan|mengetahui|kepala|nip|dokumen|status)/i.test(line)) continue
   340Ôćĺ    // Must start with a digit (attendance values)
   341Ôćĺ    if (!/^\d/.test(line)) continue
   342Ôćĺ
   343Ôćĺ    const m = line.match(attRegex)
   344Ôćĺ    if (!m) continue
   345Ôćĺ
   346Ôćĺ    const nums = m.slice(1, 16).map((n) => parseInt(n, 10))
   347Ôćĺ    // Group into 3 months: [A,I,S,JML,%]
   348Ôćĺ    const bulan: MonthAttendance[] = []
   349Ôćĺ    for (let mi = 0; mi < 3; mi++) {
   350Ôćĺ      const base = mi * 5
   351Ôćĺ      const alpa = nums[base]
   352Ôćĺ      const izin = nums[base + 1]
   353Ôćĺ      const sakit = nums[base + 2]
   354Ôćĺ      const jml = nums[base + 3]
   355Ôćĺ      const percent = nums[base + 4]
   356Ôćĺ      bulan.push({
   357Ôćĺ        nama: months[mi] || `BULAN${mi + 1}`,
   358Ôćĺ        hariEfektif: hariEfektifVals[mi] || 22,
   359Ôćĺ        alpa,
   360Ôćĺ        izin,
   361Ôćĺ        sakit,
   362Ôćĺ        jml,
   363Ôćĺ        percent,
   364Ôćĺ      })
   365Ôćĺ    }
   366Ôćĺ
   367Ôćĺ    // Extract Keterangan (Hadir / Tidak Hadir) from text after the attendance numbers
   368Ôćĺ    const afterAtt = line.substring(m.index! + m[0].length).trim()
   369Ôćĺ    let keterangan = 'Hadir'
   370Ôćĺ    const ketMatch = afterAtt.match(/(hadir|tidak\s+hadir)/i)
   371Ôćĺ    if (ketMatch) {
   372Ôćĺ      keterangan = ketMatch[1].replace(/\s+/g, ' ').trim()
   373Ôćĺ      keterangan = keterangan.charAt(0).toUpperCase() + keterangan.slice(1).toLowerCase()
   374Ôćĺ    }
   375Ôćĺ
   376Ôćĺ    // Extract Nama Pendamping: text after keterangan, uppercase words
   377Ôćĺ    let namaPendamping = ''
   378Ôćĺ    if (ketMatch && ketMatch.index !== undefined) {
   379Ôćĺ      const afterKet = afterAtt.substring(ketMatch.index + ketMatch[0].length).trim()
   380Ôćĺ      // Take uppercase words (Indonesian names are typically all-caps in these forms)
   381Ôćĺ      const nameMatch = afterKet.match(/([A-Z][A-Z\.'\s]{2,40})/)
   382Ôćĺ      if (nameMatch) {
   383Ôćĺ        namaPendamping = nameMatch[1].trim().replace(/\s+/g, ' ')
   384Ôćĺ      }
   385Ôćĺ    }
   386Ôćĺ
   387Ôćĺ    rows.push({ bulan, keterangan, namaPendamping })
   388Ôćĺ  }
   389Ôćĺ
   390Ôćĺ  return { hariEfektif: hariEfektifVals, rows }
   391Ôćĺ}

   393Ôćĺ// ---- Parse records from extracted text table ----
   394Ôćĺ// Looks for NIK (14-18 digit) patterns and builds student/participant records
   395Ôćĺexport function parseRecordsFromText(
   396Ôćĺ  text: string,
   397Ôćĺ  formType: FormType,
   398Ôćĺ  months: string[],
   399Ôćĺ  hariEfektif = DEFAULT_HARI_EFEKTIF
   400Ôćĺ): PKHRecord[] {
   401Ôćĺ  const records: PKHRecord[] = []
   402Ôćĺ  const lines = text.split(/\n/).map((l) => l.trim()).filter(Boolean)
   403Ôćĺ
   404Ôćĺ  // Pre-parse ACTUAL attendance data from the document (not random)
   405Ôćĺ  // Falls back to random only when no attendance data is found in the file
   406Ôćĺ  const { rows: attRows } = parseAttendanceFromText(text, months)
   407Ôćĺ
   408Ôćĺ  // NIK = 14-18 digit number
   409Ôćĺ  const nikRegex = /\b(\d{14,18})\b/
   410Ôćĺ
   411Ôćĺ  // Collect all NIKs in order of appearance (deduplicated)
   412Ôćĺ  const seenNiks = new Set<string>()
   413Ôćĺ
   414Ôćĺ  for (let li = 0; li < lines.length; li++) {
   415Ôćĺ    const line = lines[li]
   416Ôćĺ    // Skip NIP signature lines
   417Ôćĺ    if (/^\s*nip\.?\s/i.test(line)) continue
   418Ôćĺ    if (/^(kepala\s+sekolah|kepala\s+desa|pendamping\s+pkh|koordinator\s+pkh|mengetahui|disusun|menyetujui|catatan|dokumen|status|form|program|periode|alamat|npsn|nama\s+sekolah|direktorat|kementerian)/i.test(line)) continue
   419Ôćĺ
   420Ôćĺ    const nikMatch = line.match(nikRegex)
   421Ôćĺ    if (!nikMatch) continue
   422Ôćĺ
   423Ôćĺ    const nik = nikMatch[1]
   424Ôćĺ    if (seenNiks.has(nik)) continue
   425Ôćĺ    seenNiks.add(nik)
   426Ôćĺ
   427Ôćĺ    const beforeNik = line.substring(0, nikMatch.index || 0).trim()
   428Ôćĺ    const afterNik = line.substring((nikMatch.index || 0) + nik.length).trim()
   429Ôćĺ
   430Ôćĺ    // Extract No
   431Ôćĺ    const noMatch = beforeNik.match(/(\d+)\s*$/)
   432Ôćĺ    const no = noMatch ? parseInt(noMatch[1], 10) : records.length + 1
   433Ôćĺ
   434Ôćĺ    // Find all NIKs (14-18 digit) and NISN (10 digit) on the line
   435Ôćĺ    // Pattern from PDF: "1 3526024107900229 SOFIYATUL 3526020412090003 0095992329 MOH. QORRIFARDAN MA Kelas 10"
   436Ôćĺ    const allNiks = line.match(/\b\d{14,18}\b/g) || []
   437Ôćĺ    const allNisn = line.match(/\b\d{10}\b/g) || []
   438Ôćĺ
   439Ôćĺ    let nikPengurus = ''
   440Ôćĺ    let nikSiswa = nik
   441Ôćĺ    let nisn = ''
   442Ôćĺ    let namaPengurus = ''
   443Ôćĺ    let nama = ''
   444Ôćĺ
   445Ôćĺ    if (formType === 'education' && allNiks.length >= 2) {
   446Ôćĺ      // Structured education row: No NIKPengurus NamaPengurus NIKSiswa NISN NamaSiswa Bentuk Tingkat
   447Ôćĺ      nikPengurus = allNiks[0]
   448Ôćĺ      nikSiswa = allNiks[1]
   449Ôćĺ      // NISN is the 10-digit number after the second NIK
   450Ôćĺ      nisn = allNisn.find((n) => {
   451Ôćĺ        const pos = line.indexOf(n)
   452Ôćĺ        return pos > line.indexOf(nikSiswa)
   453Ôćĺ      }) || allNisn[0] || ''
   454Ôćĺ
   455Ôćĺ      // Nama Pengurus = text between first NIK and second NIK
   456Ôćĺ      const p1 = line.indexOf(nikPengurus)
   457Ôćĺ      const p2 = line.indexOf(nikSiswa, p1 + nikPengurus.length)
   458Ôćĺ      if (p2 > p1) {
   459Ôćĺ        namaPengurus = line.substring(p1 + nikPengurus.length, p2).trim()
   460Ôćĺ        if (namaPengurus.length > 40) namaPengurus = namaPengurus.substring(0, 40)
   461Ôćĺ      }
   462Ôćĺ
   463Ôćĺ      // Nama Siswa = text after NISN up to the Bentuk Pendidikan keyword
   464Ôćĺ      if (nisn) {
   465Ôćĺ        const nisnPos = line.indexOf(nisn, p2)
   466Ôćĺ        if (nisnPos >= 0) {
   467Ôćĺ          let afterNisn = line.substring(nisnPos + nisn.length).trim()
   468Ôćĺ          // Stop at bentuk pendidikan keywords (SD, SMP, SMA, MA, MTs, MI, PAUD, TK) or "Kelas"
   469Ôćĺ          const bentukStop = afterNisn.match(/\s+(?:SDN?|SMPN?|SMAN?|SMKN?|MTsN?|MAN?|MIN?|PAUD|TK|Madrasah)\b/i)
   470Ôćĺ          if (bentukStop && bentukStop.index !== undefined) {
   471Ôćĺ            nama = afterNisn.substring(0, bentukStop.index).trim()
   472Ôćĺ          } else {
   473Ôćĺ            // No bentuk found ÔÇö take up to "Kelas" or first digit
   474Ôćĺ            const kelasStop = afterNisn.match(/\s+Kelas\s/i)
   475Ôćĺ            if (kelasStop && kelasStop.index !== undefined) {
   476Ôćĺ              nama = afterNisn.substring(0, kelasStop.index).trim()
   477Ôćĺ            } else {
   478Ôćĺ              nama = afterNisn.split(/\s{2,}/)[0].trim()
   479Ôćĺ            }
   480Ôćĺ          }
   481Ôćĺ        }
   482Ôćĺ      }
   483Ôćĺ    } else {
   484Ôćĺ      // Single-NIK row (health/social/simple): name from previous line or after NIK
   485Ôćĺ      if (li > 0) {
   486Ôćĺ        const prevLine = lines[li - 1]
   487Ôćĺ        if (
   488Ôćĺ          prevLine.length > 2 &&
   489Ôćĺ          prevLine.length < 60 &&
   490Ôćĺ          !nikRegex.test(prevLine) &&
   491Ôćĺ          !/^\s*nip\.?\s/i.test(prevLine) &&
   492Ôćĺ          !/^(no|nama|nik|sekolah|posyandu|kecamatan|kelurahan|provinsi|kabupaten|kehadiran|pemeriksaan|bantuan|status|periode|total|kategori|formulir|kementerian|program|direktorat|catatan|hadir|tidak|qty|persentase|jenis|jumlah|alpa|izin|sakit|jml|hari|efektif|keterangan|pendamping|april|mei|juni|juli|agustus|september|oktober|november|desember|januari|februari|maret|triwulan|tahun|nisn|npsn|bentuk|pendidikan|tingkat|verifikasi|komitmen)/i.test(prevLine) &&
   493Ôćĺ          /[A-Za-z]/.test(prevLine) &&
   494Ôćĺ          !/\d{4,}/.test(prevLine)
   495Ôćĺ        ) {
   496Ôćĺ          nama = prevLine.replace(/[ÔÇö\-ÔÇó]+$/, '').trim()
   497Ôćĺ        }
   498Ôćĺ      }
   499Ôćĺ      if (!nama) {
   500Ôćĺ        const namePart = afterNik.split(/\s{2,}|\n|ÔÇö|ÔÇô/)[0] || ''
   501Ôćĺ        if (namePart.length > 2 && /[A-Za-z]/.test(namePart) && !/^\d/.test(namePart)) {
   502Ôćĺ          nama = namePart.trim()
   503Ôćĺ        }
   504Ôćĺ      }
   505Ôćĺ      if (!nama) {
   506Ôćĺ        const words = afterNik.split(/\s+/).filter((w) => /[A-Za-z]/.test(w))
   507Ôćĺ        if (words.length > 0) nama = words.slice(0, 3).join(' ')
   508Ôćĺ      }
   509Ôćĺ    }
   510Ôćĺ
   511Ôćĺ    nama = nama.replace(/[\d%]+$/, '').replace(/[ÔÇö\-ÔÇó]+$/, '').trim()
   512Ôćĺ    if (nama.length > 50) nama = nama.substring(0, 50)
   513Ôćĺ    if (!nama || nama.length < 2) continue
   514Ôćĺ
   515Ôćĺ    // Tingkat/Kelas and Bentuk Pendidikan
   516Ôćĺ    let tingkat = ''
   517Ôćĺ    let bentukPendidikan = ''
   518Ôćĺ    if (formType === 'education') {
   519Ôćĺ      const kelasMatch = line.match(/(?:kelas|kls)\s*(\d{1,2})/i)
   520Ôćĺ      if (kelasMatch) tingkat = `Kelas ${kelasMatch[1]}`
   521Ôćĺ      const bentukMatch = line.match(/\b(SDN?|SMPN?|SMAN?|SMKN?|MTs?N?|MA[N]?|MI[N]?|PAUD|TK)\b/i)
   522Ôćĺ      if (bentukMatch) bentukPendidikan = bentukMatch[1].toUpperCase()
   523Ôćĺ    }
   524Ôćĺ
   525Ôćĺ    // Posyandu (health)
   526Ôćĺ    let posyandu = ''
   527Ôćĺ    if (formType === 'health') {
   528Ôćĺ      const posMatch = line.match(/posyandu\s+\w+/i)
   529Ôćĺ      if (posMatch) posyandu = posMatch[0]
   530Ôćĺ    }
   531Ôćĺ
   532Ôćĺ    // Build the record
   533Ôćĺ    // Use ACTUAL attendance data from the document if available (matched by row order).
   534Ôćĺ    // Only fall back to random when the document has no attendance data.
   535Ôćĺ    const attIdx = records.length // 0-based index matching attendance row to student row
   536Ôćĺ    const parsedAtt = attRows[attIdx]
   537Ôćĺ    const record: PKHRecord = {
   538Ôćĺ      no,
   539Ôćĺ      nama,
   540Ôćĺ      nik: nikSiswa,
   541Ôćĺ      bulan: parsedAtt
   542Ôćĺ        ? parsedAtt.bulan
   543Ôćĺ        : months.map((m) => randomMonthAttendance(m, hariEfektif)),
   544Ôćĺ      keterangan: parsedAtt?.keterangan || 'Hadir',
   545Ôćĺ      namaPendamping: parsedAtt?.namaPendamping || '',
   546Ôćĺ    }
   547Ôćĺ
   548Ôćĺ    if (nikPengurus) record.nikPengurus = nikPengurus
   549Ôćĺ    if (namaPengurus) record.namaPengurus = namaPengurus
   550Ôćĺ    if (nisn) record.nisn = nisn
   551Ôćĺ    if (tingkat) record.tingkat = tingkat
   552Ôćĺ    if (bentukPendidikan) record.bentukPendidikan = bentukPendidikan
   553Ôćĺ    if (formType === 'education') {
   554Ôćĺ      record.sekolah = ''
   555Ôćĺ    }
   556Ôćĺ    if (formType === 'health' && posyandu) record.posyandu = posyandu
   557Ôćĺ    if (formType === 'social') {
   558Ôćĺ      record.jenisBantuan = 'PKH Reguler'
   559Ôćĺ      record.status = 'Aktif'
   560Ôćĺ    }
   561Ôćĺ
   562Ôćĺ    records.push(record)
   563Ôćĺ    if (records.length >= 200) break
   564Ôćĺ  }
   565Ôćĺ
   566Ôćĺ  return records
   567Ôćĺ}

   570Ôćĺexport async function extractFromDocument(
   571Ôćĺ  fileBuffer: Buffer,
   572Ôćĺ  filename: string
   573Ôćĺ): Promise<ParseResult> {
   574Ôćĺ  const ext = getExt(filename)
   575Ôćĺ  if (!ext) {
   576Ôćĺ    return { success: false, error: `Unsupported file type. Supported: PDF, TXT, JSON, CSV, DOCX, XLSX` }
   577Ôćĺ  }
   578Ôćĺ
   579Ôćĺ  try {
   580Ôćĺ    let text = ''
   581Ôćĺ    let wilayah: Partial<PKHFormData> = {}
   582Ôćĺ    let records: PKHRecord[] = []
   583Ôćĺ    let detectedColumns: string[] = []
   584Ôćĺ    let formType: FormType = 'social'
   585Ôćĺ
   586Ôćĺ    if (ext === 'json') {
   587Ôćĺ      const rawText = fileBuffer.toString('utf-8')
   588Ôćĺ      return parseJSONDocument(rawText)
   589Ôćĺ    }
   590Ôćĺ    if (ext === 'csv') {
   591Ôćĺ      const rawText = fileBuffer.toString('utf-8')
   592Ôćĺ      return parseCSVDocument(rawText)
   593Ôćĺ    }

   595Ôćĺ      text = fileBuffer.toString('utf-8')
   596Ôćĺ      try {
   597Ôćĺ        return parseJSONDocument(text)
   598Ôćĺ      } catch {
   599Ôćĺ        try {
   600Ôćĺ          return parseCSVDocument(text)
   601Ôćĺ        } catch {
   602Ôćĺ          // fall through to text parsing
   603Ôćĺ        }
   604Ôćĺ      }
   605Ôćĺ    } else if (ext === 'pdf') {
   606Ôćĺ      if (!existsSync(TMP_DIR)) await mkdir(TMP_DIR, { recursive: true })
   607Ôćĺ      const tmpPath = path.join(TMP_DIR, `upload-${randomUUID().slice(0, 8)}.pdf`)
   608Ôćĺ      await writeFile(tmpPath, fileBuffer)
   609Ôćĺ      try {
   610Ôćĺ        text = await extractTextFromPDF(tmpPath)
   611Ôćĺ      } finally {
   612Ôćĺ        await unlink(tmpPath).catch(() => {})
   613Ôćĺ      }
   614Ôćĺ    } else if (ext === 'docx' || ext === 'xlsx') {
   615Ôćĺ      if (!existsSync(TMP_DIR)) await mkdir(TMP_DIR, { recursive: true })
   616Ôćĺ      const tmpPath = path.join(TMP_DIR, `upload-${randomUUID().slice(0, 8)}.${ext}`)
   617Ôćĺ      await writeFile(tmpPath, fileBuffer)
   618Ôćĺ      try {
   619Ôćĺ        text = await extractTextFromOffice(tmpPath, ext)
   620Ôćĺ      } finally {
   621Ôćĺ        await unlink(tmpPath).catch(() => {})
   622Ôćĺ      }
   623Ôćĺ    }
   624Ôćĺ
   625Ôćĺ    // For PDF/TXT/DOCX/XLSX: extract wilayah + form type + records from text
   626Ôćĺ    wilayah = extractWilayahFromText(text)
   627Ôćĺ    formType = detectFormTypeFromText(text)
   628Ôćĺ
   629Ôćĺ    // Determine months from triwulan (default Triwulan 2)
   630Ôćĺ    const triwulan = wilayah.triwulan || 2
   631Ôćĺ    const months = TRIWULAN_MONTHS[triwulan] || TRIWULAN_MONTHS[2]
   632Ôćĺ    const hariEfektif = DEFAULT_HARI_EFEKTIF
   633Ôćĺ
   634Ôćĺ    records = parseRecordsFromText(text, formType, months, hariEfektif)
   635Ôćĺ
   636Ôćĺ    if (records.length === 0) {
   637Ôćĺ      return {
   638Ôćĺ        success: false,
   639Ôćĺ        error: `Tidak dapat menemukan data peserta (NIK) dalam dokumen. Pastikan dokumen berisi tabel dengan kolom NIK.`,
   640Ôćĺ        formType,
   641Ôćĺ      }
   642Ôćĺ    }
   643Ôćĺ
   644Ôćĺ    detectedColumns = extractColumnHints(text, formType)
   645Ôćĺ
   646Ôćĺ    // Build form data ÔÇö use ONLY extracted wilayah, no hardcoded defaults
   647Ôćĺ    // Facilitator fallback: use namaPendamping from first parsed attendance row
   648Ôćĺ    const facilitatorFromAtt = records.find((r) => r.namaPendamping)?.namaPendamping || ''
   649Ôćĺ    const data: PKHFormData = {
   650Ôćĺ      formType,
   651Ôćĺ      periode: wilayah.periode || `TRIWULAN 2 TAHUN ${new Date().getFullYear()}`,
   652Ôćĺ      triwulan: wilayah.triwulan || 2,
   653Ôćĺ      tahun: wilayah.tahun || new Date().getFullYear(),
   654Ôćĺ      provinsi: wilayah.provinsi || '',
   655Ôćĺ      kabupaten: wilayah.kabupaten || '',
   656Ôćĺ      kecamatan: wilayah.kecamatan || '',
   657Ôćĺ      kelurahan: wilayah.kelurahan || '',
   658Ôćĺ      npsn: wilayah.npsn || '',
   659Ôćĺ      namaSekolah: wilayah.namaSekolah || '',
   660Ôćĺ      alamatSekolah: wilayah.alamatSekolah || '',
   661Ôćĺ      signerName: wilayah.signerName || '',
   662Ôćĺ      signerNIP: wilayah.signerNIP || '',
   663Ôćĺ      signerRole: wilayah.signerRole || (formType === 'education' ? 'Kepala Sekolah' : 'Kepala Desa'),
   664Ôćĺ      facilitator: wilayah.facilitator || facilitatorFromAtt || '',
   665Ôćĺ      nipFacilitator: wilayah.nipFacilitator || '',
   666Ôćĺ      records,
   667Ôćĺ      months,
   668Ôćĺ    }
   669Ôćĺ
   670Ôćĺ    return {
   671Ôćĺ      success: true,
   672Ôćĺ      formType,
   673Ôćĺ      data,
   674Ôćĺ      detectedColumns,
   675Ôćĺ      totalRecords: records.length,
   676Ôćĺ    }
   677Ôćĺ  } catch (e) {
   678Ôćĺ    return {
   679Ôćĺ      success: false,
   680Ôćĺ      error: e instanceof Error ? e.message : 'Document extraction failed',
   681Ôćĺ    }
   682Ôćĺ  }
   683Ôćĺ}

   871Ôćĺfunction parseCSVSimple(text: string): string[][] {
   872Ôćĺ  const rows: string[][] = []
   873Ôćĺ  let current: string[] = []
   874Ôćĺ  let field = ''
   875Ôćĺ  let inQuotes = false
   876Ôćĺ
   877Ôćĺ  for (let i = 0; i < text.length; i++) {
   878Ôćĺ    const char = text[i]
   879Ôćĺ    const next = text[i + 1]
   880Ôćĺ    if (inQuotes) {
   881Ôćĺ      if (char === '"' && next === '"') { field += '"'; i++ }
   882Ôćĺ      else if (char === '"') { inQuotes = false }
   883Ôćĺ      else { field += char }
   884Ôćĺ    } else {
   885Ôćĺ      if (char === '"') { inQuotes = true }
   886Ôćĺ      else if (char === ',') { current.push(field.trim()); field = '' }
   887Ôćĺ      else if (char === '\n' || char === '\r') {
   888Ôćĺ        if (char === '\r' && next === '\n') i++
   889Ôćĺ        current.push(field.trim())
   890Ôćĺ        rows.push(current); current = []; field = ''
   891Ôćĺ      } else { field += char }
   892Ôćĺ    }
   893Ôćĺ  }
   894Ôćĺ  if (field.length > 0 || current.length > 0) {
   895Ôćĺ    current.push(field.trim())
   896Ôćĺ    rows.push(current)
   897Ôćĺ  }
   898Ôćĺ  return rows.filter((r) => r.some((c) => c.length > 0))
   899Ôćĺ}
